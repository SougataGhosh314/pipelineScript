pipeline {
    agent any
    
    stages{
        stage('clone'){
            steps{
                git 'https://github.com/SougataGhosh314/spring-boot-rest-learning.git'
            }
        }
        stage('build'){
            steps{
                parallel (
                    "Taskone" : {
                        bat 'mvn clean'
                        //do some stuff
                    },
                    "Tasktwo" : {
                        // Do some other stuff in parallel
                        bat 'mvn compile'
                    }
                )
            }
            
        }
        stage('test & QA'){
            steps{
                bat 'mvn test'
            }
        }
        stage('deploy'){
            steps{
                bat 'mvn install'    
            }
        }
    }
}
